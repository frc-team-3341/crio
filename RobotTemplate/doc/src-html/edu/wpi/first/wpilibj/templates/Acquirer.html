001package edu.wpi.first.wpilibj.templates;
002
003import edu.wpi.first.wpilibj.Jaguar;
004import edu.wpi.first.wpilibj.Talon;
005
006/**
007 * @author Spaceguy44 (Adam Tedeschi)
008 */
009public class Acquirer {
010
011    private Jaguar vandoor;
012    private Talon acquirer_left;
013    private Talon acquirer_right;
014    
015    /**
016     * Initializes the motors of the acquirer system.
017     * @param vanDoor The PWM port of the Van Door motor.
018     * @param acquirer_left The PWM port of the left acquirer motor.
019     * @param acquirer_right The PWM port of the right acquirer motor
020     */
021    public Acquirer(int vanDoor, int acquirer_left, int acquirer_right) {
022        vandoor = new Jaguar(vanDoor);
023        this.acquirer_left = new Talon(acquirer_left);
024        this.acquirer_right = new Talon(acquirer_right);
025    }
026    
027    /**
028     * Moves the acquirer towards the robot.
029     */
030    public void pitch_up () {
031        vandoor.set(0.5);  
032    }
033    
034    /**
035     * Moves the acquirer away from the robot.
036     */
037    public void pitch_down () {
038        vandoor.set(-0.5);
039    }
040    
041    /**
042     * Stops the Van Door motor from moving completely.
043     */
044    public void pitch_stop() {
045        vandoor.set(0.0);
046    }
047    
048    /**
049     * Moves the acquirer motors in opposite directions to collect the ball.
050     */
051    public void collect() {
052        acquirer_left.set(0.5);
053        acquirer_right.set(-0.5);
054    }
055    
056    /**
057     * Moves the acquirer motors in opposite directions to unload the ball.
058     */
059    public void dump() {
060        acquirer_left.set(-0.5);
061        acquirer_right.set(0.5);
062    }
063    
064    /**
065     * Stops the acquirer motors from moving completely.
066     */
067    public void acquirer_stop() {
068        acquirer_left.set(0.0);
069        acquirer_right.set(0.0);
070    }    
071}
