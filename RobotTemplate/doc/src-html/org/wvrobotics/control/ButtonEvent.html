006package org.wvrobotics.control;
007
008/**
009 *
010 * @author Vineel
011 */
012public class ButtonEvent {
013    private Controller source;
014    private int button;
015    private boolean state;
016    private long time;
017
018    public ButtonEvent(Controller source, int button, boolean state) {
019        this.source = source;
020        this.button = button;
021        this.state = state;
022        time = System.currentTimeMillis();
023    }
024
025    public Controller getSource() {
026        return source;
027    }
028
029    public int getButton() {
030        return button;
031    }
032
033    public boolean getState() {
034        return state;
035    }
036
037    public long getTime() {
038        return time;
039    }
040}
