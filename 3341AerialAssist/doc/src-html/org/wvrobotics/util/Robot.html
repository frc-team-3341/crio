001package org.wvrobotics.util;
002
003import edu.wpi.first.wpilibj.DriverStation;
004
005/**
006 * @author Vineel
007 *
008 * This class exists to allow different subsystems of the robot to communicate
009 * over the state of the robot. It is a singleton.
010 *
011 * Robot also acts as a utilities class, with functions that are universally
012 * used but do not fit into one specific place.
013 */
014public class Robot {
015    public static boolean isDisabled() {
016        return DriverStation.getInstance().isDisabled();
017    }
018
019    public static boolean isTeleop() {
020        return DriverStation.getInstance().isOperatorControl();
021    }
022
023    public static boolean isAutonomous() {
024        return DriverStation.getInstance().isAutonomous();
025    }
026
027    public static void pause(long time) {
028        try {
029            Thread.sleep(time);
030        } catch (Exception e) {}
031    }
032}
